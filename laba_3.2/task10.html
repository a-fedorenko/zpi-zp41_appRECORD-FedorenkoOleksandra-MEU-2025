Напишіть функцію checkBrackets(str) яка приймає рядок js коду (someFn) і перевіряє правильність закриття дужок () {} [] Якщо рядок містить коректний код функція повертає true. В іншому випадку повертає false

JS CODE

            function checkBrackets(str) {
                let stack = [];
                let brackets = {
                    '(': ')',
                    '{': '}',
                    '[': ']'
                };
                for (let i = 0; i < str.length; i++) {
                    if (str[i] === '(' || str[i] === '{' || str[i] === '[') {
                        stack.push(str[i]);
                    } else if (str[i] === ')' || str[i] === '}' || str[i] === ']') {
                        let last = stack.pop();
                        if (str[i] !== brackets[last]) {
                            return false;
                        }
                    }
                }
                return stack.length === 0;
            }
