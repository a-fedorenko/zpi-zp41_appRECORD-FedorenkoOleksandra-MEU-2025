            function getTags(tweets) {
                return tweets.reduce((acc, tweet) => {
                    tweet.tags.forEach(tag => {
                        acc[tag] = acc[tag] ? acc[tag] + 1 : 1;
                    });
                    return acc;
                }, {});
            }
