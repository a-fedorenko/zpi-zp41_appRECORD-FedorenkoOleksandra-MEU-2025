Напиши клас Client який створює об'єкт з ​​властивостями login email Оголоси приватні властивості #login #email, доступ до яких зроби через геттер та сеттер login email

JS CODE

            class Client {
                #login;
                #email;

                constructor(login, email) {
                    this.#login = login;
                    this.#email = email;
                }

                get login() {
                    return this.#login;
                }

                set login(value) {
                    this.#login = value;
                }

                get email() {
                    return this.#email;
                }

                set email(value) {
                    this.#email = value;
                }
            }

CONSOLE

            const client = new Client('user', 'user@gmail.com');
            console.log(`Client: ${client.login}, ${client.email}`);
            client.login = 'admin';
            client.email = 'admin@gmail.com';
            console.log(`Client: ${client.login}, ${client.email}`);
