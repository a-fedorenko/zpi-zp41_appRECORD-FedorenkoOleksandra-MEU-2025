JS CODE

            const fruits = [
                { name: "apple", price: 200 },
                { name: "orange", price: 300 },
                { name: "grapes", price: 750 },
            ];

            function discountFruits(fruits) {
                return fruits.map((fruit, index) => {
                    return {
                        ...fruit,
                        id: index + 1,
                        price: fruit.price * 0.8
                    };
                });
            }

CONSOLE

            console.log(discountFruits(fruits));
